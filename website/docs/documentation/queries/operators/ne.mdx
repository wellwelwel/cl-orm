# ne (notEqual)

> ne(column: string, param: Param)

```ts
import { OP, Param } from 'cl-orm';

/**
 * { condition: '`id` != ?', params: [16] }
 */
OP.ne('id', 16);

/**
 * { condition: '`name` != ?', params: ['John'] }
 */
OP.ne('name', 'John');
```
